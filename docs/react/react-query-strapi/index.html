Get Data with React-Query and Strapi API

Installâ€‹

yarn add @tanstack/react-query

Usageâ€‹

1. In App.tsx fileâ€‹

App.tsx
import { QueryClient, QueryClientProvider } from "@tanstack/react-query";

const queryClient = new QueryClient();

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <div className="App">
        <h1>React Query</h1>
      </div>
    </QueryClientProvider>
  );
}

2. Get Data from APIâ€‹

getData.ts
export async function getData(url: string) {
  const response = await fetch(`${url}`);
  const data = await response.json();
  return data;
}

3. In Home.tsx fileâ€‹

Home.tsx
import { useQuery } from "@tanstack/react-query";
import { getData } from "../api/getData";

export const Home = () => {
  const { data: stations } = useQuery<Root>({
    queryKey: ["stations"],
    queryFn: () =>
      getData("https://aztec-api.ktinternet.net/api/stations?populate=*"),
  });

  const image_url = "https://aztec-api.ktinternet.net";

  return (
    <div>
      <h1>Home</h1>
      {isLoading ? (
        <div>Loading...</div>
      ) : (
        <div>
          {stations?.data.map((station: Daum) => {
            const attrs: Attributes = station.attributes;

            return (
              <div
                key={station.id}
                className="h-auto m-1 overflow-hidden bg-white rounded-lg w-96"
              >
                <h2 className="text-2xl font-bold">{attrs.station_name}</h2>
                <p className="text-md">{attrs.station_tagline}</p>
                <img
                  src={
                    image_url +
                    attrs?.stationLogo?.data?.attributes?.formats?.thumbnail
                      ?.url
                  }
                  alt={attrs.station_name} // For accessibility, you might want to provide a meaningful alt text
                />
              </div>
            );
          })}
        </div>
      )}
    </div>
  );
};

4. An Example Types file for the APIâ€‹

types.ts
export interface Root {
  data: Daum[];
  meta: Meta;
}

export interface Daum {
  id: number;
  attributes: Attributes;
}

export interface Attributes {
  url: string;
  station_name: string;
  station_tagline: string;
  createdAt: string;
  updatedAt: string;
  publishedAt: string;
  stationLogo: StationLogo;
}

export interface StationLogo {
  data?: Data;
}

export interface Data {
  id: number;
  attributes: Attributes2;
}

export interface Attributes2 {
  name: string;
  alternativeText: any;
  caption: any;
  width: number;
  height: number;
  formats: Formats;
  hash: string;
  ext: string;
  mime: string;
  size: number;
  url: string;
  previewUrl: any;
  provider: string;
  provider_metadata: any;
  createdAt: string;
  updatedAt: string;
}

export interface Formats {
  large: Large;
  small: Small;
  medium: Medium;
  thumbnail: Thumbnail;
}

export interface Large {
  ext: string;
  url: string;
  hash: string;
  mime: string;
  name: string;
  path: any;
  size: number;
  width: number;
  height: number;
}

export interface Small {
  ext: string;
  url: string;
  hash: string;
  mime: string;
  name: string;
  path: any;
  size: number;
  width: number;
  height: number;
}

export interface Medium {
  ext: string;
  url: string;
  hash: string;
  mime: string;
  name: string;
  path: any;
  size: number;
  width: number;
  height: number;
}

export interface Thumbnail {
  ext: string;
  url: string;
  hash: string;
  mime: string;
  name: string;
  path: any;
  size: number;
  width: number;
  height: number;
}

export interface Meta {
  pagination: Pagination;
}

export interface Pagination {
  page: number;
  pageSize: number;
  pageCount: number;
  total: number;
}
